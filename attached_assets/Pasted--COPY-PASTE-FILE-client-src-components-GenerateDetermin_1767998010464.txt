import { useEffect, useMemo, useState } from "react";

type SupportedResponse = {
  supportedSlotIds: string[];
};

export function GenerateDeterministicButton({
  psurRef,
  slotId,
  onDone,
}: {
  psurRef: string;
  slotId: string;
  onDone?: () => void;
}) {
  const [supported, setSupported] = useState<string[] | null>(null);
  const [busy, setBusy] = useState(false);
  const [result, setResult] = useState<string>("");

  useEffect(() => {
    let cancelled = false;

    (async () => {
      try {
        const resp = await fetch("/api/slots/deterministic-supported");
        const json = (await resp.json()) as SupportedResponse;
        if (!cancelled) setSupported(json.supportedSlotIds || []);
      } catch {
        if (!cancelled) setSupported([]);
      }
    })();

    return () => {
      cancelled = true;
    };
  }, []);

  const isSupported = useMemo(() => {
    if (!supported) return false;
    return supported.includes(slotId);
  }, [supported, slotId]);

  const onClick = async () => {
    setBusy(true);
    setResult("");
    try {
      const resp = await fetch("/api/slots/generate-deterministic", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ psurRef, slotId }),
      });
      const json = await resp.json();

      if (json.adjudication === "ACCEPTED") {
        setResult(`ACCEPTED (${json.proposalId})`);
        onDone?.();
      } else {
        setResult(`REJECTED: ${(json.reasons || []).join(" | ")}`);
      }
    } catch (e: any) {
      setResult(`ERROR: ${e?.message || "unknown"}`);
    } finally {
      setBusy(false);
    }
  };

  if (supported === null) {
    // loading
    return (
      <span className="text-[10px] text-muted-foreground">
        Checking deterministic supportâ€¦
      </span>
    );
  }

  if (!isSupported) return null;

  return (
    <div className="flex items-center gap-2">
      <button
        className="px-3 py-1 rounded bg-blue-600 text-white text-xs disabled:opacity-50"
        disabled={busy}
        onClick={onClick}
      >
        {busy ? "Generating..." : "Generate (Deterministic)"}
      </button>
      {result ? (
        <span className="text-[10px] text-muted-foreground">{result}</span>
      ) : null}
    </div>
  );
}
